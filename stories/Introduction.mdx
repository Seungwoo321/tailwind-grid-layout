# Tailwind Grid Layout

A Tailwind CSS-based alternative to React Grid Layout.

## Features

- **Drag & Drop** - Drag items to change position
- **Resize** - Resize with 8-direction handles
- **Responsive** - Layout per breakpoint
- **Collision** - Collision detection and auto-placement
- **Touch Support** - Full mobile touch support

## Installation

```bash
npm install tailwind-grid-layout
# or
pnpm add tailwind-grid-layout
```

## Quick Start

```tsx
import { GridContainer } from 'tailwind-grid-layout'
import type { GridItem } from 'tailwind-grid-layout'

const items: GridItem[] = [
  { id: '1', x: 0, y: 0, w: 4, h: 2 },
  { id: '2', x: 4, y: 0, w: 4, h: 2 },
]

function App() {
  const [layout, setLayout] = useState(items)

  return (
    <GridContainer
      items={layout}
      cols={12}
      rowHeight={60}
      onLayoutChange={setLayout}
    >
      {(item) => (
        <div className="bg-blue-500 text-white p-4 rounded">
          {item.id}
        </div>
      )}
    </GridContainer>
  )
}
```
